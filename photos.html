<script>
 // Cursor Logic
 const cursor = document.getElementById('cursor');
 const trail = document.getElementById('cursorTrail');
 document.addEventListener('mousemove', e => {
   cursor.style.left = e.clientX + 'px'; cursor.style.top = e.clientY + 'px';
   setTimeout(() => { trail.style.left = e.clientX + 'px'; trail.style.top = e.clientY + 'px'; }, 60);
 });

 // DATA
 const photos = [
   { url: 'https://i.ibb.co/n8LdfSDm/IMG-20260414-055513-136.jpg', caption: 'Latest Shoot', year: '2026', location: 'Hazaribagh' },
   { url: 'https://i.ibb.co/n8LdfSDm/IMG-20260414-055513-136.jpg', caption: 'Winter Morning', year: '2025', location: 'Kota' },
   { url: 'https://i.ibb.co/n8LdfSDm/IMG-20260414-055513-136.jpg', caption: 'Greenery', year: '2024', location: 'Village' },
   { url: 'https://i.ibb.co/n8LdfSDm/IMG-20260414-055513-136.jpg', caption: 'City Lights', year: '2023', location: 'Ranchi' },
   { url: 'https://i.ibb.co/n8LdfSDm/IMG-20260414-055513-136.jpg', caption: 'Childhood Home', year: 'Old', location: 'Village' },
   { url: 'https://i.ibb.co/n8LdfSDm/IMG-20260414-055513-136.jpg', caption: 'School Days', year: '2022', location: 'School' },
 ];

 let currentFilter = { year: 'all', location: 'all' };

 function renderPhotos() {
   const grid = document.getElementById('photoGrid');
   const filtered = photos.filter(p => {
     return (currentFilter.year === 'all' || p.year === currentFilter.year) &&
            (currentFilter.location === 'all' || p.location === currentFilter.location);
   });

   document.getElementById('photoCount').textContent = `${filtered.length} photos`;
   document.getElementById('emptyState').style.display = filtered.length ? 'none' : 'block';

   grid.innerHTML = filtered.map((p, i) => `
     <div class="photo-item" style="animation-delay: ${i * 0.05}s" onclick="openLightbox('${p.url}', '${p.caption}', '${p.year}', '${p.location}')">
       <img src="${p.url}" alt="${p.caption}" loading="lazy">
       <div class="photo-overlay">
         <div class="photo-caption">${p.caption}</div>
         <div class="photo-meta">
           <span class="photo-tag tag-year">📅 ${p.year}</span>
           <span class="photo-tag tag-location">📍 ${p.location}</span>
         </div>
       </div>
     </div>
   `).join('');
 }

 function filterPhotos(type, value) {
   currentFilter[type] = value;
   const btns = event.target.parentNode.querySelectorAll('.ftab');
   btns.forEach(b => b.classList.remove('active-tab'));
   event.target.classList.add('active-tab');
   renderPhotos();
 }

 function openLightbox(url, cap, yr, loc) {
   document.getElementById('lightboxImg').src = url;
   document.getElementById('lightboxCaption').textContent = cap;
   document.getElementById('lightboxMeta').innerHTML = `<span class="photo-tag tag-year">📅 ${yr}</span> <span class="photo-tag tag-location">📍 ${loc}</span>`;
   document.getElementById('lightbox').classList.add('open');
 }

 function closeLightbox() { document.getElementById('lightbox').classList.remove('open'); }

 // Initial Render
 renderPhotos();
</script>
